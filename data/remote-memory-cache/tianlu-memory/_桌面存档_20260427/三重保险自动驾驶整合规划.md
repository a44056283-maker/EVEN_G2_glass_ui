# 三重保险自动驾驶整合规划

> **更新时间**: 2026-04-25
> **目标**: 天眼AI + 出山AI + Bot备用自动驾驶 = 三重保险，永不错过入场/出场机会

---

## 一、问题背景与整合目标

### 1.1 当前痛点

| 问题 | 现状 | 影响 |
|------|------|------|
| Bot数据可能有偏差 | Bot自动驾驶依赖Bot端采集的数据，可能与M1/M2/M3不一致 | 入场时机偏差 |
| 两套系统独立运行 | 天眼/出山跑9099，Bot自动驾驶跑Bot端 | 重复下单或漏单 |
| 无失效检测 | 无法判断天眼/出山是否正常运行 | 失效时无感知 |
| 无自动切换 | Bot自动驾驶需手动启动/停止 | 运维复杂 |

### 1.2 整合目标

```
整合后的架构（三重保险）

第1重: 天眼AI自动驾驶（主系统）
├── 运行位置: 9099 console_server
├── 数据源: M1/M2/M3直连交易所
├── 决策: AI评测 + 规则引擎
└── 执行: Bot RPC

第2重: 出山AI自动驾驶（主系统）
├── 运行位置: 9099 console_server
├── 数据源: M1/M2/M3直连交易所
├── 决策: AI评测 + 规则引擎
└── 执行: Bot RPC

第3重: Bot备用自动驾驶（备用系统）
├── 运行位置: Bot端 v65_autopilot.py
├── 数据源: Bot端数据（可能偏差）
├── 决策: 纯规则引擎
└── 执行: Bot内部RPC
└── 触发条件: 天眼/出山失效时自动激活
```

```python
# console_server.py 新增

def _health_monitor_loop():
    """
    健康检测线程（每60秒运行）
    检测天眼/出山是否失效，自动切换Bot备用
    """
    global _CONTROL_MODE, _BOT_FALLBACK_ACTIVE

    consecutive_recovery = 0  # 连续恢复次数

    while True:
        time.sleep(60)

        tianyan_dead = (
            time.time() - _TIANYAN_LAST_HEARTBEAT > _HEARTBEAT_TIMEOUT
        )
        exit_dead = (
            time.time() - _EXIT_AI_LAST_HEARTBEAT > _HEARTBEAT_TIMEOUT
        )

        # === 检测失效：切换到Bot备用 ===
        if not _BOT_FALLBACK_ACTIVE:
            if tianyan_dead or exit_dead:
                reason = []
                if tianyan_dead:
                    reason.append("天眼")
                if exit_dead:
                    reason.append("出山")
                _activate_bot_fallback(", ".join(reason))

        # === 检测恢复：切回天眼/出山 ===
        else:
            both_alive = (
                time.time() - _TIANYAN_LAST_HEARTBEAT < _HEARTBEAT_TIMEOUT
                and time.time() - _EXIT_AI_LAST_HEARTBEAT < _HEARTBEAT_TIMEOUT
            )
            if both_alive:
                consecutive_recovery += 1
                if consecutive_recovery >= _RECOVERY_THRESHOLD:
                    _deactivate_bot_fallback()
                    consecutive_recovery = 0
            else:
                consecutive_recovery = 0
```

### 4.3 自动切换逻辑

```python
# console_server.py 新增

_CONTROL_MODE = "tianyan"    # "tianyan" | "bot_fallback"
_BOT_FALLBACK_ACTIVE = False

def _activate_bot_fallback(reason: str):
    """
    激活Bot备用自动驾驶
    触发条件: 天眼/出山心跳超时
    """
    global _CONTROL_MODE, _BOT_FALLBACK_ACTIVE
    _CONTROL_MODE = "bot_fallback"
    _BOT_FALLBACK_ACTIVE = True

    # 1. 停止天眼/出山扫描
    _tianyan_log(f"[⚠️ 系统切换] {reason}失效，激活Bot备用自动驾驶")

    # 2. 通过RPC启动所有Bot的备用自动驾驶
    for port in ALL_PORTS:
        try:
            r = requests.post(
                f"{_get_rpc_url(port)}/api/v1/autopilot/start",
                json={"mode": "fallback", "source": "tianyan_dead"},
                auth=_get_bot_auth(port), timeout=10
            )
            if r.status_code != 200:
                _tianyan_log(f"[⚠️ Bot备用激活失败] port={port} {r.text}")
        except Exception as e:
            _tianyan_log(f"[⚠️ Bot备用激活异常] port={port} {e}")

    # 3. 写入共享状态文件
    _write_control_mode_state()

    # 4. 飞书通知
    _send_feishu_alert(
        f"⚠️ 系统切换: {reason}失效\n"
        f"Bot备用自动驾驶已激活\n"
        f"时间: {_format_time()}"
    )

def _deactivate_bot_fallback():
    """
    停用Bot备用自动驾驶，切回天眼/出山
    触发条件: 天眼/出山心跳恢复正常（连续3次）
    """
    global _CONTROL_MODE, _BOT_FALLBACK_ACTIVE
    _CONTROL_MODE = "tianyan"
    _BOT_FALLBACK_ACTIVE = False

    _tianyan_log("[✅ 系统恢复] 天眼/出山已恢复，切回天眼/出山控制")

    # 1. 停止所有Bot的备用自动驾驶
    for port in ALL_PORTS:
        try:
            requests.post(
                f"{_get_rpc_url(port)}/api/v1/autopilot/stop",
                json={"mode": "fallback", "reason": "tianyan_recovered"},
                auth=_get_bot_auth(port), timeout=10
            )
        except Exception as e:
            _tianyan_log(f"[⚠️ Bot备用停用失败] port={port} {e}")

    # 2. 写入共享状态文件
    _write_control_mode_state()

    # 3. 飞书通知
    _send_feishu_alert(
        f"✅ 系统恢复: 天眼/出山已恢复\n"
        f"Bot备用自动驾驶已停用\n"
        f"时间: {_format_time()}"
    )

def _write_control_mode_state():
    """写入控制模式状态到共享文件"""
    state = {
        "mode": _CONTROL_MODE,
        "bot_fallback_active": _BOT_FALLBACK_ACTIVE,
        "tianyan_heartbeat": _TIANYAN_LAST_HEARTBEAT,
        "exit_ai_heartbeat": _EXIT_AI_LAST_HEARTBEAT,
        "switch_time": time.time(),
    }
    with open("/tmp/tianlu_control_mode.json", "w") as f:
        json.dump(state, f, indent=2)
```
