## 六、机器人快速查询命令

```bash
# 批量查询余额
for port in 9090 9091 9092 9093 9094 9095 9096 9097; do
  auth="freqtrade:freqtrade"
  if [ "$port" -ge "9093" ]; then auth="admin:Xy@06130822"; fi
  balance=$(curl -s -u "$auth" "http://localhost:$port/api/v1/balance" 2>/dev/null | python3 -c "import sys,json;d=json.load(sys.stdin);print(f\"\${d.get('total',0):.2f}\")")
  echo "Port $port: \$$balance"
done

# 批量查询持仓
for port in 9090 9091 9092 9093 9094 9095 9096 9097; do
  auth="freqtrade:freqtrade"
  if [ "$port" -ge "9093" ]; then auth="admin:Xy@06130822"; fi
  curl -s -u "$auth" "http://localhost:$port/api/v1/status" 2>/dev/null | python3 -c "
import sys,json;data=json.load(sys.stdin)
if not data: print(f'Port $port: 空仓')
for d in data:
    pnl=(d.get('profit_ratio') or 0)*100
    print(f\"Port $port: {d['pair']}({'空' if d.get('is_short') else '多'}){pnl:+.2f}%\")
" 2>/dev/null
done
```
