# 对比分析（compare.html）完整设计规范
> 版本：v3.1 | 日期：2026-04-15 | 给天福：明天直接用这个文件开发
> 风格：同步持仓评测（live-eval）视觉风格

---

## 一、页面定位

- **名称**：对比分析（compare.html）
- **路径**：`/freqtrade_console/static/tabs/compare.html`
- **数据源**：`/api/backtest/all_trades?limit=5000`（当前），`/api/backtest/compare`（未来后端）
- **定位**：持仓评测（live-eval）的横向视角，多端口对比，识别哪个端口的入场/出场系统更优

---

## 二、页面布局（从上到下）

```
┌──────────────────────────────────────────────────────────┐
│  top-bar（固定顶栏）                                       │
│  [🔍 对比分析] | [Bot卡片×8横向滚动] | [⚡分析对比] [状态] │
├──────────────────────────────────────────────────────────┤
│  stat-bar（统计6格）                                      │
│  [端口数] [总交易] [冠军] [胜率] [均收益] [资金费率]        │
├──────────────────────────────────────────────────────────┤
│  content（可滚动内容区）                                   │
│  ├── rank-section（排行榜）                               │
│  ├── tri-grid（三栏：入场/止盈/出场）                      │
│  ├── 盈亏分布柱图                                        │
│  ├── 资金费率柱图                                        │
│  ├── 汇局限论                                            │
│  └── port-cards（端口详细卡片，可展开话术）                │
└──────────────────────────────────────────────────────────┘
```

### 12.2 综合评分公式

```javascript
function calcCompositeScore(stats) {
  const winRateScore = Math.min(stats.wins / stats.total * 100, 100) * 0.30;
  const avgProfitScore = Math.max(0, Math.min((stats.avgProfit + 5) / 10 * 100, 100)) * 0.35;
  const entryScore = stats.estL1PassRate * 0.20;
  const exitScore = Math.max(0, (100 - stats.bigLoss / stats.total * 5)) * 0.15;
  return Math.round(Math.min(winRateScore + avgProfitScore + entryScore + exitScore, 99));
}
```

### 12.4 端口统计计算（calcPortStats）

```javascript
function calcPortStats(trades, port) {
  const closed = trades.filter(t => t.close_date && t.profit_ratio != null);
  if (!closed.length) return null;

  const profits = closed.map(t => t.profit_ratio * 100);
  const wins = profits.filter(p => p > 0);
  const bigWin = profits.filter(p => p > 5).length;
  const medWin = profits.filter(p => p > 1 && p <= 5).length;
  const smallWin = profits.filter(p => p > 0 && p <= 1).length;
  const smallLoss = profits.filter(p => p >= -5 && p < 0).length;
  const bigLoss = profits.filter(p => p < -5).length;

  // 估算L1量比通过率
  let volPass = 0, volTotal = 0;
  let longCount = 0, shortCount = 0;
  for (const t of closed) {
    const tag = (t.enter_tag || '').toLowerCase();
    const isLong = !tag.includes('short');
    if (isLong) longCount++; else shortCount++;
    const m = tag.match(/vol(\d+(?:\.\d+)?)/i);
    if (m) { volTotal++; if (parseFloat(m[1]) >= V65.LONG_VOL_MIN) volPass++; }
  }
  const estL1PassRate = volTotal > 0 ? volPass / volTotal * 100 : 60;
  const estL2PassRate = Math.min(estL1PassRate * 0.92, 95);
  const estL4PassRate = Math.min(estL1PassRate * 0.85, 90);

  // 动态止盈P1/P2/P3触发率
  const p1Trades = profits.filter(p => p >= 15 && p < 20).length;
  const p2Trades = profits.filter(p => p >= 20 && p < 25).length;
  const p3Trades = profits.filter(p => p >= 25).length;
  const estP1Rate = closed.length > 0 ? p1Trades / closed.length * 100 : 0;
  const estP2Rate = closed.length > 0 ? p2Trades / closed.length * 100 : 0;
  const estP3Rate = closed.length > 0 ? p3Trades / closed.length * 100 : 0;

  // 资金费率
  const fundingCost = closed.reduce((s,t) => s + Math.abs(t.funding_fees || 0), 0);

  // 综合评分
  const compositeScore = calcCompositeScore({
    wins: wins.length, total: closed.length,
    avgProfit: profits.reduce((s,p)=>s+p,0) / closed.length,
    bigLoss, estL1PassRate
  });

  return {
    port, exchange: EXCHANGES[port],
    count: closed.length, winRate: wins.length / closed.length * 100,
    avgProfit: profits.reduce((s,p)=>s+p,0) / closed.length,
    fundingCost, fundingCostPerTrade: closed.length > 0 ? fundingCost / closed.length : 0,
    bigWin, medWin, smallWin, smallLoss, bigLoss,
    estL1PassRate, estL2PassRate, estL4PassRate,
    estP1Rate, estP2Rate, estP3Rate,
    compositeScore,
  };
}
```
