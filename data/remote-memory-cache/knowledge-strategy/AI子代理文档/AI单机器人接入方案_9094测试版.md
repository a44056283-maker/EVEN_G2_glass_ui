# ai_9094_agent.py
"""
AI单机器人代理（9094专用）
Phase 1: 黑名单过滤
"""
import time
import sqlite3
from datetime import datetime
from pathlib import Path

DB_PATH = Path(__file__).parent.parent / "rules_config" / "ai_9094.db"

class AI9094Agent:
    def __init__(self):
        self._init_db()
        self.blacklist = self._load_blacklist()
        self.market_structure_cache = {}
        self.cache_ttl = 300  # 5分钟

    def _init_db(self):
        """初始化SQLite"""
        conn = sqlite3.connect(DB_PATH, check_same_thread=False)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS blacklist (
                pair TEXT PRIMARY KEY,
                level TEXT DEFAULT 'none',  -- none/watch/warn/block
                consecutive_losses INTEGER DEFAULT 0,
                win_rate REAL DEFAULT 0.0,
                updated_at TEXT
            )
        """)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS alert_log (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                ts TEXT,
                port INTEGER,
                pair TEXT,
                alert_type TEXT,  -- wrong_direction/premature
                action TEXT,      -- half_exit/full_exit/keep
                result TEXT,      -- success/fail/suppressed
                pnl_after_10min REAL,
                is_accurate INTEGER  -- 1/0/NULL
            )
        """)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS ai_decisions (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                ts TEXT,
                port INTEGER,
                pair TEXT,
                decision TEXT,  -- allowed/blocked/warn
                reason TEXT,
                ai_score REAL
            )
        """)
        conn.commit()
        return conn

    def is_blocked(self, pair):
        """检查黑名单"""
        pair = pair.split(":")[0]  # 处理 "BTC/USDT:USDT" 格式
        level = self.blacklist.get(pair, {}).get("level", "none")
        return level in ("warn", "block")

    def record_trade_result(self, pair, is_win, pnl_pct):
        """交易结束后记录结果，更新黑名单"""
        conn = sqlite3.connect(DB_PATH, check_same_thread=False)
        cur = conn.execute(
            "SELECT consecutive_losses, total_trades FROM blacklist WHERE pair=?",
            (pair,)
        )
        row = cur.fetchone()
        if row:
            losses, total = row
            new_losses = 0 if is_win else losses + 1
            new_total = total + 1
            win_rate = (total - losses if is_win else total - new_losses) / new_total
            level = self._calc_level(new_losses, win_rate)
            conn.execute(
                "UPDATE blacklist SET consecutive_losses=?, win_rate=?, level=?, updated_at=? WHERE pair=?",
                (new_losses, win_rate, level, datetime.now().isoformat(), pair)
            )
        else:
            conn.execute(
                "INSERT INTO blacklist (pair, consecutive_losses, win_rate, level, updated_at) VALUES (?, ?, ?, ?, ?)",
                (pair, 0 if is_win else 1, 1.0 if is_win else 0.0, "none", datetime.now().isoformat())
            )
        conn.commit()
        self.blacklist = self._load_blacklist()

    def _calc_level(self, losses, win_rate):
        if losses >= 5 or win_rate < 0.25:
            return "block"
        elif losses >= 3 or win_rate < 0.35:
            return "warn"
        elif losses >= 2:
            return "watch"
        return "none"

    def should_block(self, pair, direction):
        """AI决策：是否阻止开仓"""
        pair = pair.split(":")[0]
        info = self.blacklist.get(pair, {})
        level = info.get("level", "none")

        if level == "block":
            return True, f"{pair} 在黑名单中，禁止开仓"
        if level == "warn":
            return True, f"{pair} 在警告名单，胜率{info.get('win_rate',0)*100:.0f}%，建议跳过"
        if level == "watch":
            return False, f"{pair} 在观察名单，+20分入场门槛"
        return False, "通过"

    # ── Phase 2: 市场雷达 ───────────────────────────────
    def get_market_structure(self, pair):
        """获取市场结构（带缓存）"""
        if pair in self.market_structure_cache:
            cached, ts = self.market_structure_cache[pair]
            if time.time() - ts < self.cache_ttl:
                return cached
        # TODO: 调用MiniMax API
        # 暂时返回默认
        return {"structure": "range", "confidence": 60, "trend": "neutral"}

    def should_boost_entry_threshold(self, pair, direction):
        """市场结构不利时，加严入场门槛"""
        struct = self.get_market_structure(pair)
        if struct["structure"] == "range" and direction == "long":
            return True, "震荡市场做多，入场门槛+20分"
        if struct["structure"] == "strong_down" and direction == "long":
            return True, "强下跌趋势，不建议逆势做多"
        return False, "通过"

    # ── Phase 3: AI动态止盈 ─────────────────────────────
    def get_dynamic_exit_mult(self, pair, direction, base_mult):
        """AI动态调整止盈倍数"""
        struct = self.get_market_structure(pair)
        if struct["structure"] == "range":
            # 震荡行情，止盈下调20%
            return base_mult * 0.8, "震荡市，止盈下调20%"
        if struct["structure"] == "strong_trend":
            # 强趋势，止盈放宽30%
            return base_mult * 1.3, "强趋势，止盈放宽30%"
        return base_mult, "正常"

# 全局单例
